vector<int> Mob(int N) {
    vector<int> mob(N + 1, 0);
    for (int i = 2; i <= N; i++) {
        if (mob[i] != 0)continue;
        for (int j = i; j <= N; j += i)mob[j]++;
        for (long long j = (ll)i * i; j <= N; j += i * i)mob[j] = -1e9;
    }
    for (int i = 0; i <= N; i++) {
        if (mob[i] < 0)mob[i] = 0;
        else if (mob[i] % 2)mob[i] = -1;
        else mob[i] = 1;
    }
    return mob;
}
