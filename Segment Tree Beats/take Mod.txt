const int N = 4e5;
vector<pair<int,int> > tree(N);
// sum max
pair<int,int> merge(pair<int,int> a, pair<int,int> b) {
    return {a.first + b.first, max(a.second, b.second)};
}

void assign(int pos,int idx,int val,int st,int en) {
    if (idx < st || idx > en)return;
    int mid = st + (en - st) / 2;
    if (st == en) {
        tree[pos] = {val, val};
        return;
    }
    assign(pos * 2 + 1,idx, val, st, mid);
    assign(pos * 2 + 2,idx, val, mid + 1, en);
    tree[pos] = merge(tree[pos * 2 + 1], tree[pos * 2 + 2]);
}

void mod(int pos,int val,int l,int r,int st,int en) {
    if (l > en || r < st || tree[pos].second < val)return;
    if (st == en) {
        tree[pos] = {tree[pos].first % val, tree[pos].first % val};
        return;
    }
    int mid = st + (en - st) / 2;
    mod(pos * 2 + 1, val, l, r, st, mid);
    mod(pos * 2 + 2, val, l, r, mid + 1, en);
    tree[pos] = merge(tree[pos * 2 + 1], tree[pos * 2 + 2]);
}

int query(int pos,int l,int r,int st,int en) {
    if (l > en || r < st)return 0;
    if (st>=l&&en<=r) {
        return tree[pos].first;
    }
    int mid = st + (en - st) / 2;
    return query(pos * 2 + 1, l, r, st, mid) + query(pos * 2 + 2, l, r, mid + 1, en);
}

    for (int i = 0; i < n; i++)cin >> vals[i], assign(0,i, vals[i], 0, n - 1);
    while (q--) {

        }
        else {

            assign(0,k,x,0,n-1);
        }
    }
}
