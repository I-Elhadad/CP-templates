const int N = 2e6;

struct item {
    int mx=LONG_LONG_MIN;
    int secMx = LONG_LONG_MIN;
    int freqMx = 0;
    int sum=0;
    int lazyMn = LONG_LONG_MIN;
    int lazyAdd = 0;
};
vector<item> tree(N);
// sum max

item merge(item a, item b) {
    item ret{0, 0, 0, 0};
    ret.mx = max(a.mx, b.mx);
    if (a.mx == b.mx) {
        ret.freqMx = a.freqMx + b.freqMx;
        ret.secMx = max(b.secMx, a.secMx);
    } else {
        if (a.mx < b.mx)
            swap(a, b);
        ret.freqMx = a.freqMx;
        ret.secMx = max(b.mx, a.secMx);
    }
    ret.sum = a.sum + b.sum;
    ret.lazyAdd = 0;
    ret.lazyMn=LONG_LONG_MIN;
    return ret;
}

void updateMn(int pos,int val) {
    if (val >= tree[pos].mx)return;
    tree[pos].sum -= (tree[pos].freqMx * (tree[pos].mx - val));
    tree[pos].mx = val;
    if (tree[pos].lazyMn == LONG_LONG_MIN) tree[pos].lazyMn = val;
    else tree[pos].lazyMn = min(tree[pos].lazyMn, val);
}

void updateAdd(int pos,int val,int l,int r) {
    tree[pos].sum += (r - l + 1) * val;
    tree[pos].mx += val;
    if (tree[pos].secMx!=LONG_LONG_MIN)
    tree[pos].secMx += val;
    tree[pos].lazyAdd += val;
    if (tree[pos].lazyMn != LONG_LONG_MIN) tree[pos].lazyMn += val;

}

void propagate(int pos,int l,int r) {
    if (tree[pos].lazyMn == LONG_LONG_MIN && !tree[pos].lazyAdd)return;
    if (l != r) {
        int valAdd = tree[pos].lazyAdd;
        int valMn = tree[pos].lazyMn;
        int mid = (l + r) / 2;
        if (valAdd != 0) {
            updateAdd(pos * 2 + 1, valAdd, l, mid);
            updateAdd(pos * 2 + 2, valAdd, mid + 1, r);
        }
        if (valMn != LONG_LONG_MIN) {
            updateMn(pos * 2 + 1, valMn);
            updateMn(pos * 2 + 2, valMn);
        }
    }
    tree[pos].lazyMn = LONG_LONG_MIN;
    tree[pos].lazyAdd = 0;
}

void assign(int pos,int idx,int val,int st,int en) {
    if (idx < st || idx > en)return;
    int mid = st + (en - st) / 2;
    if (st == en) {
        tree[pos] = {val, LONG_LONG_MIN, 1, val,LONG_LONG_MIN,0};
        return;
    }
    assign(pos * 2 + 1, idx, val, st, mid);
    assign(pos * 2 + 2, idx, val, mid + 1, en);
    tree[pos] = merge(tree[pos * 2 + 1], tree[pos * 2 + 2]);
}

void Min(int pos,int val,int l,int r,int st,int en) {
    propagate(pos, st, en);
    if ((l > en || r < st || tree[pos].mx <= val))return;
    if ((st >= l && en <= r) && tree[pos].secMx < val) {
        updateMn(pos, val);
        return;
    }
    if (st == en) {
        tree[pos].mx = min(tree[pos].mx, val);
        tree[pos].sum = tree[pos].mx;
        return;
    }
    int mid = st + (en - st) / 2;
    Min(pos * 2 + 1, val, l, r, st, mid);
    Min(pos * 2 + 2, val, l, r, mid + 1, en);
    tree[pos] = merge(tree[pos * 2 + 1], tree[pos * 2 + 2]);
}

void Add(int pos,int val,int l,int r,int st,int en) {
    propagate(pos, st, en);
    if (l > en || r < st)return;
    if ((st >= l && en <= r)) {
        updateAdd(pos, val, st, en);
        return;
    }
    int mid = st + (en - st) / 2;
    Add(pos * 2 + 1, val, l, r, st, mid);
    Add(pos * 2 + 2, val, l, r, mid + 1, en);
    tree[pos] = merge(tree[pos * 2 + 1], tree[pos * 2 + 2]);
}

int query(int pos,int l,int r,int st,int en) {
    propagate(pos, st, en);
    if (l > en || r < st)return 0;
    if (st >= l && en <= r) {
        return tree[pos].sum;
    }
    int mid = st + (en - st) / 2;
    return query(pos * 2 + 1, l, r, st, mid) + query(pos * 2 + 2, l, r, mid + 1, en);
}
