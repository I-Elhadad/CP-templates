struct Manacher {
    vector<int> p;

    void runManacher(string s) {
        int l = 1, r = 1;
        int n = s.size();
        p.assign(n, 1);
        for (int i = 1; i < n; i++) {
            p[i] = max(0, min(r - i, p[l + r - i]));
            while (i + p[i] < n && i - p[i] >= 0 && s[i + p[i]] == s[i - p[i]]) {
                p[i]++;
            }
            // #a#b#c#
            if (i + p[i] > r) {
                l = i - p[i];
                r = i + p[i];
            }
        }
    }

    void build(string s) {
        string t = "";
        for (auto it: s) {
            t += string("#") + it;
        }
        t += '#';
        runManacher(t);
    }
    // odd equal 1 mean cen is center of palin (ODD LEN)
    // odd equal 0 mean cen is  left center of palin (EVEN LEN)
    int getLongest(int cen, bool odd) {
        int pos = cen * 2 + 1 + (!odd);
        return p[pos] - 1;
    }

    bool checkPalin(int l, int r) {
        return (r - l + 1 <= getLongest((l + r) / 2, (l % 2 == r % 2)));
    }
} m;
